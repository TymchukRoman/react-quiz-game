## Install

```bash
npm install --save react-quiz-game
```

## Usage

```tsx
import React, { Component } from 'react'

import { Quiz } from 'react-quiz-game'
import 'react-quiz-game/dist/index.css'

class Example extends Component {
  render() {
    return <Quiz props />
  }
}
```
